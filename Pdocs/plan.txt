  Frontend currently does:
    GET /api/feed → 20 posts
    GET /api/system/reaction/system-like/ID1  (for every post)

Code logic in getHomeFeed:
  const userId = req.userId; // may be undefined for guests

  // For user posts — userReaction check:
  userReaction: userId
    ? (p.likedBy?.some(id => id.toString() === userId) ? 'like'
      : p.dislikedBy?.some(id => id.toString() === userId) ? 'dislike'
      : null)
    : null,

  // For system posts — extend reactionMap lookup:
  userReaction: userId
    ? (reactionMap[post._id]?.likedBy?.some(id => id.toString() === userId)

    // In-memory cache (15 minutes TTL — same pattern as JSON posts cache)
    if (_trendingCache && Date.now() - _trendingCacheTime < 15 * 60 * 1000) {
      return res.json({ success: true, data: _trendingCache });

    _trendingCacheTime = Date.now();
    return res.json({ success: true, data: trending });

STEP 2.7 — useFeed Hook (Infinite Scroll via Intersection Observer)
File: src/hooks/useFeed.js

Replace "Load More" button with automatic scroll detection.

const useFeed = () => {
  const [posts, setPosts] = useState([]);
  const [page, setPage] = useState(1);
  const [hasMore, setHasMore] = useState(true);
  const [loading, setLoading] = useState(false);
  const loaderRef = useRef(null);

  const fetchFeed = useCallback(async (pageNum) => {
    setLoading(true);
    const res = await feedService.getFeed(pageNum);
    // reactions now embedded in feed — no N+1 requests needed
    if (pageNum === 1) setPosts(res.data);
    else setPosts(prev => [...prev, ...res.data]);
    setHasMore(res.hasNextPage);
    setLoading(false);
  }, []);

  useEffect(() => {
    const observer = new IntersectionObserver(
      entries => {
        if (entries[0].isIntersecting && hasMore && !loading) {
          setPage(prev => prev + 1);
        }
      },
      { threshold: 0.1 }
    );
    if (loaderRef.current) observer.observe(loaderRef.current);
    return () => observer.disconnect();
  }, [hasMore, loading]);

  useEffect(() => { fetchFeed(page); }, [page]);

  return { posts, loading, hasMore, loaderRef, setPosts };
};

STEP 2.8 — useSearch Hook (Debounced Live Search)
File: src/hooks/useSearch.js

const useSearch = () => {
  const [query, setQuery] = useState('');
  const [results, setResults] = useState(null);
  const [isOpen, setIsOpen] = useState(false);
  const [history, setHistory] = useState(
    () => JSON.parse(localStorage.getItem('searchHistory') || '[]')
  );

  useEffect(() => {
    if (!query.trim()) { setResults(null); return; }
    const timer = setTimeout(async () => {
      const res = await searchService.getSuggestions(query);
      setResults(res);
      setIsOpen(true);
    }, 350); // 350ms debounce
    return () => clearTimeout(timer);
  }, [query]);

  const addToHistory = (term) => {
    const updated = [term, ...history.filter(h => h !== term)].slice(0, 5);
    setHistory(updated);
    localStorage.setItem('searchHistory', JSON.stringify(updated));
  };

  return { query, setQuery, results, isOpen, setIsOpen, history, addToHistory };
};

STEP 6.2 — Relative Time Utility
File: src/utils/formatDate.js

  export const timeAgo = (dateStr) => {
    const date = new Date(dateStr);
    if (isNaN(date)) return '';
    const diff = Date.now() - date.getTime();
    const minutes = Math.floor(diff / 60000);
    if (minutes < 1) return 'just now';
    if (minutes < 60) return `${minutes}m`;
    const hours = Math.floor(minutes / 60);
    if (hours < 24) return `${hours}h`;
    const days = Math.floor(hours / 24);
    if (days < 7) return `${days}d`;
    return date.toLocaleDateString('en-US', { month: 'short', day: 'numeric' });
  };

STEP 6.3 — Avatar Initials Fallback Component
File: src/components/shared/Avatar.jsx

  const COLORS = ['#6366f1', '#8b5cf6', '#ec4899', '#14b8a6', '#f59e0b'];

  const Avatar = ({ src, name, size = 40 }) => {
    const [imgError, setImgError] = useState(false);
    const initials = name?.split(' ').map(w => w[0]).join('').slice(0,2).toUpperCase() || '?';
    const color = COLORS[(name?.charCodeAt(0) || 0) % COLORS.length];

    if (!src || imgError) {
      return (
        <div style={{ width: size, height: size, backgroundColor: color }}
             className="rounded-full flex items-center justify-center text-white font-semibold text-sm">
          {initials}
        </div>
      );
    }
    return <img src={src} alt={name} width={size} height={size}
                className="rounded-full object-cover"
                onError={() => setImgError(true)} />;
  };

  export const formatCount = (n) => {
    if (!n) return '0';
    if (n >= 1_000_000) return `${(n / 1_000_000).toFixed(1)}M`;
    if (n >= 1_000) return `${(n / 1_000).toFixed(1)}K`;

================================================================================
  ARCHITECTURE DECISIONS — WHY EACH CHOICE WAS MADE
================================================================================

1. WHY split Feed.jsx into 5 files?
   → Single Responsibility Principle. 561 lines = untestable, unscalable.
   → FeedCard can be reused in Profile, Explore, Search results.
   → Recruiter reviewing code sees modular thinking.

2. WHY fix N+1 at backend vs frontend?
   → Frontend cannot batch reaction fetches efficiently.
   → Backend does single MongoDB $in query for all postIds at once.
   → Result: 20 HTTP requests → 1. ~95% reduction.

3. WHY Intersection Observer for infinite scroll?
   → Twitter, Instagram, LinkedIn all use this pattern.
   → Users scroll naturally — no extra click required.
   → useInfiniteScroll.js hook is reusable across multiple pages.

4. WHY pre-compute AI score on creation vs on feed load?
   → Computing AI score on every feed load = expensive API calls
     (N posts × 3-8 second AI latency = unusable feed).
   → Pre-compute once on creation, store result → feed always fast.
   → Fire-and-forget: post creation response is NOT delayed.

5. WHY in-memory cache for trending (not Redis)?
   → Redis adds ops complexity and cost for a portfolio project.
   → In-memory 15-minute cache matches existing JSON posts cache.
   → Consistent, easy to upgrade to Redis later.

6. WHY localStorage for search history?
   → No backend needed — instant read, zero API cost.
   → Works even when logged out.
   → Same approach as Twitter/X search history.

7. WHY move Logout to Settings dropdown?
   → Destructive actions should not be prominently placed (UX principle).
   → Users stop accidentally logging out.
   → Twitter, LinkedIn, GitHub all follow this exact pattern.

8. WHY Avatar initials fallback?
   → Broken image icons destroy platform trust immediately.
   → Color derived from name = each user has unique consistent color.
   → Same pattern: Google, Notion, Linear, Vercel.

9. WHY deduplicate tags?
   → Live production bug found: #india #india #india on a post.
   → Set() dedup is O(n), trivially prevents DB clutter.
   → Cap at 15 tags to prevent spam.

10. WHY line-clamp-3 on feed cards?
    → Users scan feeds — wall of text = scroll fatigue = churn.
    → Twitter, Medium, LinkedIn all truncate feed previews.
    → Full content accessible on post detail page.
